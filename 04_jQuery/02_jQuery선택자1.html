기본 속성 선택자

        기본 속성 선택자(css) : 선택자[속성="속성명"]

        [JS]
        document.querySelector("기본속성 선택자"); // 일치하는 맨 처음 요소 1개
        document.querySelectorAll("기본속성 선택자"); // 일치하는 요소 모두

        [jQuery]
        $("기본 속성 선택자"); // 일치하는 모든 요소
